CREATE TABLE Resume (
	id int NOT NULL AUTO_INCREMENT,
	name varchar(255) NOT NULL,
	email varchar(255) NOT NULL,
	phone varchar(255) NOT NULL,
	skills TEXT,
	experience TEXT,
	education TEXT,
	awards TEXT,
	honors TEXT,
	languages TEXT NOT NULL,
	personal_references TEXT,
	interests TEXT,
	technology TEXT,
	certification TEXT,
	projects TEXT,
	summary TEXT,
	objective TEXT,
	PRIMARY KEY (id)
);

CREATE TABLE Job_Description (
	id int NOT NULL AUTO_INCREMENT,
	job_title TEXT NOT NULL,
	description longtext NOT NULL,
	skills TEXT NOT NULL,
	experience TEXT NOT NULL,
	education TEXT NOT NULL,
	languages TEXT NOT NULL,
	location TEXT NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE Weightage (
	id int NOT NULL AUTO_INCREMENT,
	job_description_id int NOT NULL,
	skills_weightage FLOAT NOT NULL DEFAULT 0.25,
	education_weightage FLOAT NOT NULL DEFAULT 0.25,
	languages_weightage FLOAT NOT NULL DEFAULT 0.25,
	experience_weightage FLOAT NOT NULL DEFAULT 0.25,
	PRIMARY KEY (id)
);

ALTER TABLE Weightage ADD CONSTRAINT Weightage_fk0 FOREIGN KEY (job_description_id) REFERENCES Job_Description(id);
